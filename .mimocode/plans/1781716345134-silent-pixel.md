# Деплой LaserCut Core на продакшн

## Рекомендуемый стек

| Компонент | Технология |
|-----------|------------|
| Сервер | VPS 4 CPU / 8 GB RAM / 80 GB SSD |
| ОС | Ubuntu 22.04 LTS |
| Контейнеры | Docker + Docker Compose |
| Reverse Proxy | Nginx (в Docker или хостовый) |
| SSL | Let's Encrypt (certbot) |
| Бэкапы | pg_dump + cron + S3/Minio |
| Мониторинг | Uptime Kuma + docker logs |
| Домен | Любой (.ru, .com, .site) |

## Процесс деплоя (пошагово)

### Шаг 1: Подготовка сервера

```bash
# Подключаемся к серверу
ssh root@YOUR_SERVER_IP

# Обновляем систему
apt update && apt upgrade -y

# Устанавливаем Docker
curl -fsSL https://get.docker.com -o get-docker.sh
sh get-docker.sh

# Устанавливаем Docker Compose
apt install docker-compose-plugin -y

# Создаём директорию проекта
mkdir -p /opt/lasercut
cd /opt/lasercut
```

### Шаг 2: Копирование проекта

```bash
# На локальном компьютере
git clone git@github.com:YOUR_REPO/laser-cut-backend.git
cd laser-cut-backend

# Копируем на сервер
scp -r . root@YOUR_SERVER_IP:/opt/lasercut/

# Или через git на сервере
cd /opt/lasercut
git clone git@github.com:YOUR_REPO/laser-cut-backend.git .
```

### Шаг 3: Настройка окружения

```bash
# На сервере
cd /opt/lasercut

# Создаём .env файл
cat > .env << 'EOF'
DATABASE_URL=postgresql+asyncpg://postgres:СЛОЖНЫЙ_ПАРОЛЬ@db:5432/laser_cut
REDIS_URL=redis://redis:6379/0
SECRET_KEY=СГЕНЕРИРУЙТЕ_СЛУЧАЙНУЮ_СТРОКУ_32_СИМВОЛА
ENVIRONMENT=prod
EOF

# Генерируем секретный ключ
python3 -c "import secrets; print(secrets.token_hex(32))"
```

### Шаг 4: Настройка docker-compose.prod.yml

```yaml
services:
  frontend:
    build:
      context: ./frontend
    restart: unless-stopped
    depends_on:
      - backend

  backend:
    build: .
    command: uvicorn app.main:app --host 0.0.0.0 --port 8000 --workers 4
    volumes:
      - ./data/images:/app/data/images
      - ./data/uploads:/app/data/uploads
    env_file:
      - .env
    environment:
      - DATABASE_URL=postgresql+asyncpg://postgres:ПАРОЛЬ@db:5432/laser_cut
      - SECRET_KEY=ВАШ_КЛЮЧ
    depends_on:
      - db
      - redis
    restart: unless-stopped
    deploy:
      resources:
        limits:
          memory: 2G

  db:
    image: postgres:15-alpine
    environment:
      POSTGRES_USER: postgres
      POSTGRES_PASSWORD: СЛОЖНЫЙ_ПАРОЛЬ
      POSTGRES_DB: laser_cut
    volumes:
      - postgres_data:/var/lib/postgresql/data
    restart: unless-stopped
    deploy:
      resources:
        limits:
          memory: 1G

  redis:
    image: redis:7-alpine
    command: redis-server --maxmemory 256mb --maxmemory-policy allkeys-lru
    volumes:
      - redis_data:/data
    restart: unless-stopped

  nginx:
    image: nginx:alpine
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./nginx.prod.conf:/etc/nginx/conf.d/default.conf
      - ./certbot/conf:/etc/letsencrypt
      - ./certbot/www:/var/www/certbot
    depends_on:
      - frontend
      - backend
    restart: unless-stopped

  certbot:
    image: certbot/certbot
    volumes:
      - ./certbot/conf:/etc/letsencrypt
      - ./certbot/www:/var/www/certbot
    entrypoint: "/bin/sh -c 'trap exit TERM; while :; do certbot renew; sleep 12h & wait $${!}; done;'"

volumes:
  postgres_data:
  redis_data:
```

### Шаг 5: Настройка Nginx

```nginx
# nginx.prod.conf
server {
    listen 80;
    server_name lasercut.yourdomain.com;

    location /.well-known/acme-challenge/ {
        root /var/www/certbot;
    }

    location / {
        return 301 https://$host$request_uri;
    }
}

server {
    listen 443 ssl http2;
    server_name lasercut.yourdomain.com;

    ssl_certificate /etc/letsencrypt/live/lasercut.yourdomain.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/lasercut.yourdomain.com/privkey.pem;

    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers HIGH:!aNULL:!MD5;

    client_max_body_size 50M;

    location / {
        proxy_pass http://frontend:80;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    location /auth/ {
        proxy_pass http://backend:8000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
    }

    location /users/ {
        proxy_pass http://backend:8000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
    }

    location /audit/ {
        proxy_pass http://backend:8000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
    }

    location /orders/ {
        proxy_pass http://backend:8000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
    }

    location /api/ {
        proxy_pass http://backend:8000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
    }

    location /ws/ {
        proxy_pass http://backend:8000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
        proxy_read_timeout 86400;
    }

    location /health {
        proxy_pass http://backend:8000;
    }
}
```

### Шаг 6: Получение SSL-сертификата

```bash
# Сначала запускаем без SSL для получения сертификата
docker compose up -d nginx

# Получаем сертификат
docker compose run --rm certbot certonly \
  --webroot \
  --webroot-path=/var/www/certbot \
  --email admin@yourdomain.com \
  -d lasercut.yourdomain.com \
  --agree-tos \
  --no-eff-email

# Перезапускаем всё
docker compose down
docker compose up -d
```

### Шаг 7: Запуск

```bash
# Сборка и запуск
docker compose -f docker-compose.prod.yml up -d --build

# Проверка статуса
docker compose ps

# Проверка логов
docker compose logs -f backend
```

## Бэкапы

### Автоматический бэкап PostgreSQL

```bash
# Создаём скрипт бэкапа
cat > /opt/lasercut/backup.sh << 'EOF'
#!/bin/bash
DATE=$(date +%Y-%m-%d_%H-%M)
BACKUP_DIR="/opt/lasercut/backups"
mkdir -p $BACKUP_DIR

# Бэкап БД
docker exec laser-cut-backend-db-1 pg_dump -U postgres laser_cut | gzip > $BACKUP_DIR/db_$DATE.sql.gz

# Удаляем старые бэкапы (старше 7 дней)
find $BACKUP_DIR -name "*.sql.gz" -mtime +7 -delete

echo "Backup completed: $DATE"
EOF

chmod +x /opt/lasercut/backup.sh

# Добавляем в cron (каждый день в 3:00)
crontab -e
# 0 3 * * * /opt/lasercut/backup.sh >> /var/log/lasercut-backup.log 2>&1
```

### Восстановление из бэкапа

```bash
# Восстановление
gunzip < backups/db_2026-06-17_03-00.sql.gz | docker exec -i laser-cut-backend-db-1 psql -U postgres laser_cut
```
